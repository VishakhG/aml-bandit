      }
      
      \ELSE
      \STATE{

        Set $\tilde{f_t} \leftarrow ELLIPSOIDSAMPLE(x_t, \tilde{\mu_t})$
      }
      \ENDIF \\
    $x_{t+1} = argmin_{x \in \mathcal{K}}[\eta \sum_{\tau =1}^t \tilde{f_{\tau}^T} x + \mathcal{R}(x)]$
    }
    \ENDFOR
